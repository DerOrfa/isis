72std::vector<ValueArray> ValueArray::splice(size_t size) const{
73 if ( size >= getLength() ) {
74 LOG( Debug, warning )
75 << "splicing data of the size " << getLength() << " up into blocks of the size " << size << " is kind of useless ...";
76 }
77
78 const size_t fullSplices = getLength() / size;
79 const size_t lastSize = getLength() % size;//rest of the division - size of the last spliceAt
80 const size_t splices = fullSplices + ( lastSize ? 1 : 0 );
81
82 _internal::DelProxy proxy( *this );
83
84 auto generator = [&](auto ptr){
85 std::vector<ValueArray> ret(splices );
86
87 for ( size_t i = 0; i < fullSplices; i++ )
88 ret[i]=ValueArray(ptr.get() + i * size, size, proxy );
89
90 if ( lastSize )
91 ret.back()= ValueArray(ptr.get() + fullSplices * size, lastSize, proxy );
92
93 return ret;
94 };
95
96 return std::visit(generator,static_cast<const ArrayTypes&>(*this));
97}

179{

249std::size_t ValueArray::compare(std::size_t start, std::size_t end, const ValueArray& dst, std::size_t dst_start) const
250{
251 assert( start <= end );
252 size_t ret = 0;
253 size_t _length = end - start;
254
255 if ( dst.getTypeID() != getTypeID() ) {
256 LOG( Debug, error )
257 << "Comparing to a ValueArray of different type(" << dst.typeName() << ", not " << typeName()
258 << "). Assuming all voxels to be different";
259 return _length;
260 }
261
262 LOG_IF( end >= getLength(), Runtime, error )
263 << "End of the range (" << end << ") is behind the end of this ValueArray (" << getLength() << ")";
264 LOG_IF( _length + dst_start >= dst.getLength(), Runtime, error )
265 << "End of the range (" << _length + dst_start << ") is behind the end of the destination (" << dst.getLength() << ")";
266
267 // lock the memory, so we can mem-compare the elements (use uint8_t because some compilers do not like arith on void*)
268 const std::shared_ptr<const uint8_t>
269 src_s = std::static_pointer_cast<const uint8_t>( getRawAddress() ),
270 dst_s = std::static_pointer_cast<const uint8_t>( dst.getRawAddress() );
271 const uint8_t *src_p = src_s.get(), *dst_p = dst_s.get();
272 const size_t el_size = bytesPerElem();
273
274 for ( size_t i = start; i < end; i++ ) {
275 if ( memcmp( src_p + ( i * el_size ), dst_p + ( i * el_size ), el_size ) != 0 )
276 ret++;
277 }
278
279 return ret;
280}
