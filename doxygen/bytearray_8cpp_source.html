30void make_generators(std::map<unsigned short, generator_fn>& map) { //The ID here is the index in ArrayTypes, NOT the isis-typeID

34 map[index]=[]( data::ByteArray &bytes, size_t offset, size_t len, bool swap_endianess ){

36 };
