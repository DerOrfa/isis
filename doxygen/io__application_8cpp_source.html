161std::list< Image > IOApplication::autoload ( const util::ParameterMap &parameters, bool exitOnError, const std::string &suffix, std::shared_ptr<util::ProgressFeedback> feedback, util::slist* rejected)
162{
163 const util::slist input = parameters[std::string( "in" ) + suffix];
164 const util::slist rf = parameters[std::string( "rf" ) + suffix];
165 const util::slist dl = parameters[std::string( "rdialect" ) + suffix];
166
167 const bool no_progress = parameters["np"];
168
169 if( !no_progress && feedback ) {
170 data::IOFactory::setProgressFeedback( feedback );
171 }
172
173 const std::list<util::istring> formatstack=util::makeIStringList(rf);
174
175 std::list< Image > tImages;
176 if(input.size()==1 && input.front()=="-"){
177 LOG( Runtime, info )
178 << "loading from stdin"
179 << util::NoSubject( rf.empty() ? "" : std::string( " using the format stack: " ) + util::listToString(rf.begin(),rf.end()) )
180 << util::NoSubject( ( !rf.empty() && !dl.empty() ) ? " and" : "" )
181 << util::NoSubject( dl.empty() ? "" : std::string( " using the dialect: " ) + util::listToString(dl.begin(),dl.end()) );
182 tImages = data::IOFactory::load( std::cin.rdbuf(), formatstack, util::makeIStringList(dl),rejected );
183 } else {
184 LOG( Runtime, info )
185 << "loading " << util::MSubject( input )
186 << util::NoSubject( rf.empty() ? "" : std::string( " using the format stack: " ) + util::listToString(rf.begin(),rf.end()) )
187 << util::NoSubject( ( !rf.empty() && !dl.empty() ) ? " and" : "" )
188 << util::NoSubject( dl.empty() ? "" : std::string( " using the dialect: " ) + util::listToString(dl.begin(),dl.end()) );
189 tImages = data::IOFactory::load( input, formatstack, util::makeIStringList(dl),rejected );
190 }
191
192 if ( tImages.empty() ) {
193 if ( exitOnError ) {
194 LOG( Runtime, notice ) << "No data acquired, exiting...";
195 exit( 1 );
196 }
197 } else {
198 for( auto a = tImages.begin(); a != tImages.end(); a++ ) {
199 for( auto b = a; ( ++b ) != tImages.end(); ) {
200 const data::Image &aref = *a, bref = *b;
201 LOG_IF( aref.getDifference( bref ).empty(), Runtime, warning ) << "The metadata of the images "
202 << aref.identify(true,false) << ":" << std::distance<std::list<Image> ::const_iterator>( tImages.begin(), a )
203 << " and " << bref.identify(true,false) << ":" << std::distance<std::list<Image> ::const_iterator>( tImages.begin(), b )
204 << " are equal. Maybe they are duplicates.";
205 }
206 }
207 }
208
209 return tImages;
210}

223bool IOApplication::autowrite ( const util::ParameterMap &parameters, std::list< Image > out_images, bool exitOnError, const std::string &suffix)
224{
225 std::optional<util::Selection> repn;
226 if(parameters[std::string( "repn" ) + suffix].isParsed())
227 repn = parameters[std::string( "repn" ) + suffix].as<util::Selection>();
228
229 const std::string output = parameters[std::string( "out" ) + suffix];
230 const util::slist wf = parameters[std::string( "wf" ) + suffix];
231 const util::slist dl = parameters[std::string( "wdialect" ) + suffix];
232 LOG( Runtime, info )
233 << "Writing " << out_images.size() << " images"
234 << ( repn ? std::string( " as " ) + static_cast<std::string>(*repn) : "" )
235 << " to " << util::MSubject( output )
236 << ( wf.empty() ? "" : std::string( " using the format: " ) + util::listToString(wf.begin(),wf.end()) )
237 << ( ( !wf.empty() && !dl.empty() ) ? " and" : "" )
238 << ( dl.empty() ? "" : std::string( " using the dialect: " ) + util::listToString(dl.begin(),dl.end()) );
239
240 if( repn ) {
241 for( std::list<Image>::reference ref : out_images ) {
242 ref.convertToType( static_cast<unsigned short>(*repn) );
243 }
244 }
245
246 if( feedback() )
247 data::IOFactory::setProgressFeedback( feedback() );
248
249 if ( ! IOFactory::write( out_images, output, util::makeIStringList(wf), util::makeIStringList(dl) ) ) {
250 if ( exitOnError ) {
251 LOG( Runtime, notice ) << "Failed to write, exiting...";
252 exit( 1 );
253 }
254 return false;
255 } else
256 return true;
257}
