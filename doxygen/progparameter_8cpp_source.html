39bool ProgParameter::parse( const std::string &prop )
40{
41 Value &me = this->front();
42 bool ret = false;
43
44 if ( prop.empty() ) {
45 if ( me.is<bool>() ) {
46 std::get<bool>(me) = true;
47 ret = true;
48 }
49 } else {
50 ret = Value::convert(prop, me );
51 }
52
53 LOG_IF( ret, Debug, info ) << "Parsed " << MSubject( prop ) << " as " << me.toString( true );
54
55 if( ret ) {
56 setNeeded(false);//remove needed flag, because the value is set (aka "not needed anymore")
57 m_parsed = true;
58 }
59
60 return ret;
61}

101bool ParameterMap::parse( int argc, char **argv )
102{
103 parsed = true;
104 std::string pName;
105
106 for ( int i = 1; i < argc; ) { // scan through the command line, to find next parameter
107 if ( argv[i][0] == '-' && argv[i][1]!=0 ) { //seems like we found a new parameter here
108 pName = argv[i];
109 pName.erase( 0, pName.find_first_not_of( '-' ) );
110 } else {
111 LOG( Runtime, error ) << "Ignoring unexpected non-parameter " << MSubject( argv[i] );
112 }
113
114 i++;
115
116 if( !pName.empty() ) { // if we got a parameter before
117 const int start = i;
118
119 std::list<std::string> matchingStrings;
120 for( ParameterMap::const_reference parameterRef : *this ) {
121 if( parameterRef.first.find( pName ) == 0 ) {
122 if( parameterRef.first.length() == pName.length() ) { //if it's an exact match
123 matchingStrings = std::list<std::string>( 1, pName ); //use that
124 break;// and stop searching for partial matches
125 } else
126 matchingStrings.push_back( parameterRef.first );
127 }
128 }
129
130 if( matchingStrings.size() > 1 ) { // multiple matches
131 std::stringstream matchingStringStream;
132 for( std::list<std::string>::const_reference stringRef : matchingStrings ) {
133 matchingStringStream << stringRef << " ";
134 }
135 LOG( Runtime, warning )
136 << "The parameter \"" << pName << "\" is ambiguous. The parameters \""
137 << matchingStringStream.str().erase( matchingStringStream.str().size() - 1, 1 )
138 << "\" are possible. Ignoring this parameter!";
139 continue;
140 } else if ( matchingStrings.empty() ) { // no match
141 LOG( Runtime, warning ) << "Ignoring unknown parameter " << MSubject( std::string( "-" ) + pName + " " + listToString( argv + start, argv + i, " ", "", "" ) );
142 continue;
143 }
144
145 // exact one match
146 ProgParameter &match=at(matchingStrings.front() );
147
148 // check if parameter can be empty (bool only)
149 if(match.is<bool>()){ //no parameters expected
150 match.parse(""); // ok that's it nothing else to do here
151 } else { //go through all parameters
152 i++;//skip the first one, that should be here (enables us to have parameters like "-20")
153 while( i < argc && !(argv[i][0] == '-' && argv[i][1] != 0) ) {
154 i++;//collect the remaining parameters, while there are some ..
155 }
156
157 parsed = match.is<util::slist>() ? //don't do tokenizing if the target is a slist (is already done by the shell)
158 match.parse_list( util::slist( argv + start, argv + i ) ) :
159 match.parse( listToString( argv + start, argv + i, ",", "", "" ) );
160 LOG_IF(!parsed, Runtime, error )
161 << "Failed to parse the parameter " << MSubject( std::string( "-" ) + matchingStrings.front() ) << ": "
162 << ( start == i ? "nothing" : listToString( argv + start, argv + i, " ", "\"", "\"" ) )
163 << " was given, but a " << match.getTypeName() << " was expected.";
164 }
165
166 }
167 }
168
169 return parsed ;
170}

178{

180
