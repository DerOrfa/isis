26template<typename SRC, typename DST> void numeric_convert_impl( const SRC *src, DST *dst, size_t count, double scale, double offset )
27{
28 LOG( Runtime, info )
29 << "using generic scaling convert " << util::typeName<SRC>() << "=>" << util::typeName<DST>()
30 << " with scale/offset " << std::fixed << scale << "/" << offset;
31
32 static const std::pair<DST, DST> limits(
33 std::numeric_limits<DST>::max(),
34 std::numeric_limits<DST>::has_denorm ? -std::numeric_limits<DST>::max() : std::numeric_limits<DST>::min() //for types with denormalization min is _not_ the lowest value
35 );
36
37 for ( size_t i = 0; i < count; i++ ) {
38 auto scaled=src[i] * scale + offset;
39 dst[i] =
40 scaled<limits.second ?
41 limits.second : scaled > limits.first ?
42 limits.first : round<DST>( scaled );
43 }
44}

205template<typename SRC, typename DST> scaling_pair getNumericScaling(const util::Value &min, const util::Value &max )
206{
207 double scale = 1.0;
208 double offset = 0.0;
209
210 if ( std::is_integral_v<DST> ) {//only do scale if the target is an integer (scaling into float is useless)
211 const DST domain_min = std::numeric_limits<DST>::min();//negative value domain of this dst [min .. -1]
212 const DST domain_max = std::numeric_limits<DST>::max();//positive value domain of this dst [0 .. max]
213 double minval, maxval;
214 minval = min.as<double>();
215 maxval = max.as<double>();
216 LOG_IF(std::isnan(minval) && std::isnan(maxval), Runtime, error) << "The min as well as the max of the range is NaN, thats not going to end well...";
217 assert( minval <= maxval );
218 LOG( Debug, info ) << "src Range:" << minval << "=>" << maxval;
219 LOG( Debug, info ) << "dst Domain:" << static_cast<double>( domain_min ) << "=>" << static_cast<double>( domain_max );
220 assert( domain_min < domain_max );//we also should assume this
221
222 //set offset for src
223 //if all src is completly on positive domain, or if there is no negative domain use minval
224 //else if src is completly on negative dmain, or if there is no positive domain use maxval
225 //elsewise leave it at 0 and scale both sides
226 if ( minval > 0 || !domain_min ) {
227 // if ( ( maxval - domain_max ) > 0 ) // if the values completely fit into the domain we dont have to offset them
228 offset = -minval;
229 } else if ( ( 0 - maxval ) > 0 || !domain_max ) {
230 // if ( ( domain_min - minval ) > 0 ) // if the values completely fit into the domain we dont have to offset them
231 offset = -maxval;
232 }
233
234 //calculate range of values which will be on postive/negative domain when offset is applied
235 const double range_max = maxval + offset; //allways >=0
236 const double range_min = minval + offset; //allways <=0
237 //set scaling factor to fit src-range into dst domain
238 //some compilers dont make x/0 = inf, so we use std::numeric_limits<double>::max() instead, in this case
239 const double scale_max =
240 range_max != 0 ? domain_max / range_max :
241 std::numeric_limits<double>::max();
242 const double scale_min =
243 range_min != 0 ? domain_min / range_min :
244 std::numeric_limits<double>::max();
245 scale = std::min( scale_max ? scale_max : std::numeric_limits<double>::max(), scale_min ? scale_min : std::numeric_limits<double>::max() );//get the smaller scaling factor which is not zero so the bigger range will fit into his domain
246
247 if ( std::is_integral_v<SRC> && scale > 1 ) {
248 LOG( Debug, verbose_info ) << "Limiting scaling to 1, because the source datatype " << util::typeName<SRC>() << " is discrete";
249 scale = 1;
250 }
251
252 if( scale == 1 ) {
253 if( ( minval - domain_min ) > 0 && ( domain_max - maxval ) > 0 )
254 offset = 0; // if the source does fit into the domain, and we do not scale - we wont need an offset
255 } else
256 offset *= scale;//calc offset for dst
257 }
258
259 return { scale, offset };
260}
