97bool Application::addConfigFile(const std::string& filename)
98{
99 data::FilePtr f(filename);
100 if(f.good()){
101 const data::ValueArray buffer=f.at<uint8_t>(0);
102 if(configuration.readJson(buffer.beginTyped<uint8_t>(),buffer.endTyped<uint8_t>(),'/')==0){
103 auto param=configuration.queryBranch("parameters");
104 // if there is a "parameters" section in the file, use that as default parameters for the app
105 if(param){
106 for(PropertyMap::PropPath p:param->getLocal<PropertyValue>()){
107 assert(p.size()==1);
108 ProgParameter &dst=parameters[p.front().c_str()];
109 PropertyValue &src=param->touchProperty(p);
110 if(!dst.isParsed()){ // don't touch it, if It's not just the default
111 LOG(Runtime,verbose_info) << "Setting parameter " << std::make_pair(p,src) << " from configuration";
112 if(dst.isEmpty())
113 dst.swap(src);
114 else if(!dst.front().apply(src.front())){ //replace builtin default with value from config if there is one already
115 LOG(Runtime,warning) << "Failed to apply parameter " << std::make_pair(p,src) << " from configuration, skipping ..";
116 continue;
117 }
118 }
119 dst.setNeeded(false); //param has got its default from the configuration, so its not needed anymore
120 param->remove(p);
121 }
122 }
123 return true;
124 } else {
125 LOG(Runtime,warning) << "Failed to parse configuration file " << util::MSubject(filename);
126 }
127 }
128 return false;
129}

138bool Application::initBase( int argc, char **argv )
139{
140 m_filename=argv[0];
141 if ( !parameters.parse( argc, argv ) ){
142 LOG( Runtime, error ) << "Failed to parse the command line";
143 return false;
144 }
145
146 if ( parameters["help"] ) {
147 printHelp( true );
148 exit( 0 );
149 }
150
151 auto cfg=parameters.find("cfg");
152 if(cfg!=parameters.end()){
153 if(!addConfigFile(cfg->second.as<std::string>()))
154 return false;
155 }
156
157 resetLogging();
158
159 if(!parameters["np"])
160 feedback() = std::make_shared<util::ConsoleProgressBar>();
161
162
163 const std::string loc=parameters["locale"];
164 if(!loc.empty())
165 std::setlocale(LC_ALL,loc.c_str());
166
167 return true;
168}

170bool Application::init( int argc, char **argv, bool exitOnError )
171{
172 bool good = initBase(argc,argv);
173
174 if ( ! parameters.isComplete() ) {
175 std::cerr << "Missing parameters: ";
176
177 for ( const auto &P: parameters ) {
178 if ( P.second.isNeeded() ) {std::cerr << "-" << P.first << " ";}
179 }
180
181 std::cerr << std::endl;
182 printHelp();
183 good = false;
184 }
185
186 if ( !good && exitOnError ) { //we did printHelp already above
187 std::cerr << "Exiting..." << std::endl;
188 exit( 1 );
189 }
190
191 return good;
192}
193std::list<std::shared_ptr<MessageHandlerBase>> Application::resetLogging()
194{
195 std::list<std::shared_ptr<MessageHandlerBase>> ret;
196 for( auto &ref: logs ) {
197 const std::string dname = "d"s + ref.first;
198 assert( !parameters[dname].isEmpty() ); // this must have been set by addLoggingParameter (called via addLogging)
199 auto level = ( LogLevel )static_cast<uint8_t>(parameters[dname].as<Selection>());
200 for( setLogFunction setter: ref.second ) {
201 ret.push_back(std::move((this->*setter )( level )));
202 }
203 }
204 return std::move(ret);
205}

207void Application::printHelp( bool withHidden )const
208{
209 typedef std::list<std::pair<std::string, std::string> >::const_reference example_type;
210 std::cerr << this->m_name << " (using isis " << getCoreVersion() << ")" << std::endl;
211 std::cerr << "Usage: " << this->m_filename << " <options>" << std::endl << "Where <options> includes:" << std::endl;;
212
213 for (const auto & parameter : parameters) {
214 std::string pref;
215
216 if ( parameter.second.isNeeded() ) {
217 pref = ". Required.";
218 } else if( parameter.second.isHidden() ) {
219 if( !withHidden )
220 continue; // if its hidden, not needed, and wie want the short version skip this parameter
221 }
222
223 if ( ! parameter.second.isNeeded() ) {
224 pref = ". Default: \"" + parameter.second.toString() + "\".";
225 }
226
227 std::cerr
228 << "\t-" << parameter.first << " <" << parameter.second.getTypeName() << ">" << std::endl
229 << "\t\t" << parameter.second.description() << pref << std::endl;
230
231 if ( parameter.second.is<Selection>() ) {
232 const Selection &ref = parameter.second.as<Selection>();
233 const std::list< istring > entries = ref.getEntries();
234 auto i = entries.begin();
235 std::cerr << "\t\tOptions are: \"" << *i << "\"";
236
237 for( i++ ; i != entries.end(); i++ ) {
238 auto dummy = i;
239 std::cout << ( ( ++dummy ) != entries.end() ? ", " : " or " ) << "\"" << *i << "\"";
240 }
241
242 std::cerr << "." << std::endl;
243 }
244 }
245
246 if( !m_examples.empty() ) {
247 std::cout << "Examples:" << std::endl;
248
249 for( example_type ex : m_examples ) {
250 std::cout << '\t' << m_filename << " " << ex.first << '\t' << ex.second << std::endl;
251 }
252 }
