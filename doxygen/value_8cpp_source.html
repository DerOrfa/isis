11template<typename T1, typename T2> concept three_way_comparable_with = requires(T1 &&v1, T2 &&v2){{ v1 <=> v2 } -> std::convertible_to<std::partial_ordering>;};
12template<typename T1, typename T2> concept three_way_comparable_with_non_value = non_value<T1> && non_value<T2> && three_way_comparable_with<T1, T2>;
13template<typename T1, typename T2> concept equal_comparable_with = requires(T1 &&v1, T2 &&v2){{ v1 == v2 } -> std::convertible_to<bool>;};;
14template<typename T1, typename T2> concept equal_comparable_with_non_value = non_value<T1> && non_value<T2> && equal_comparable_with<T1, T2>;

20std::partial_ordering static_three_way_compare(const Value &lhs, const three_way_comparable_non_value auto &rhs)
21{
22 typedef std::remove_cvref_t<decltype(rhs)> r_type;
23 auto visitor = [&rhs](const auto &lhs) -> std::partial_ordering
24 {
25 typedef std::remove_cvref_t<decltype(lhs)> l_type;
26 //@todo std::compare_partial_order_fallback once it's widely available
27 if constexpr(three_way_comparable_with<l_type, r_type>)
28 return lhs <=> rhs;
29 else if constexpr(std::is_convertible_v<r_type,l_type> && three_way_comparable_with<l_type, l_type>)
30 return lhs <=> l_type(rhs);
31 else if constexpr(std::is_convertible_v<l_type,r_type>) // we wouldn't be in here if three_way_comparable_with<r_type> wasn't true
32 return r_type(lhs) <=> rhs;
33 else
34 return std::partial_ordering::unordered;
35 };
36 auto result = std::visit(visitor, static_cast<const ValueTypes &>(lhs));
37 LOG_IF(result==std::partial_ordering::unordered,Runtime, error)
38 << "Cannot compare " << lhs.toString(true) << " and "
39 << rhs << "(" << _internal::typename_with_fallback<r_type>() << ")";
40
41 return result;
42}
43bool static_equal_compare(const Value &lhs, const equal_comparable_non_value auto &rhs)
44{

46 auto visitor = [&](auto &&ptr) -> bool

78template<class OP> [[nodiscard]] Value inner_arithmetic_op(const Value &l_value,const auto &rhs)
79{
80 static const OP op;
81 auto visitor = [&rhs](const auto &lhs) -> Value
82 {
83 typedef std::remove_cvref_t<decltype(lhs)> l_type;
84 typedef std::remove_cvref_t<decltype(rhs)> r_type;
85
86 if constexpr(op_available_non_value<OP,l_type,r_type>) {
87 auto result = op(lhs, rhs);
88 if constexpr(KnownValueType<decltype(result)>)
89 return result;
90 else
91 LOG(Runtime,error)
92 << "The result of the operation " << typeid(op).name() << " on "
93 << util::typeName<l_type>() << " and " << typename_with_fallback<r_type>()
94 << "(" << typename_with_fallback<decltype(result)>() << ") cannot be represented as Value. It will be ignored.";
95 } else
96 LOG(Runtime,error)
97 << "Invalid operation " << typeid(op).name() << " on " << util::typeName<l_type>()
98 << " and " << typename_with_fallback<r_type>();
99 return lhs;
100 };
101 return std::visit(visitor,static_cast<const ValueTypes&>(l_value));

103template<class OP> [[nodiscard]] Value arithmetic_op(const Value &l_value,const Value& r_value)
