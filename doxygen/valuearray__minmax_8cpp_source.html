65template<typename T> std::pair<__m128i, __m128i> _getMinMaxBlockLoop( const __m128i *data, size_t blocks )
66{
67 std::pair<__m128i, __m128i> ret( _mm_loadu_si128( data ), _mm_loadu_si128( data ) );
68 LOG( Runtime, verbose_info ) << "using optimized min/max computation for " << util::typeName<T>() << " (masked mode)";
69 static const __m128i one = _mm_set_epi16( -1, -1, -1, -1, -1, -1, -1, -1 );
70
71 while ( --blocks ) {
72 const __m128i at = _mm_loadu_si128( ++data );
73
74 const __m128i less_mask = _TypeVector<T>::lt( at, ret.first );
75 const __m128i greater_mask = _TypeVector<T>::gt( at, ret.second );
76
77 ret.first =
78 ( ret.first & ( less_mask ^ one ) ) //remove bigger values from current min
79 | ( at & less_mask ); //put in the lesser values from at
80 ret.second =
81 ( ret.second & ( greater_mask ^ one ) ) //remove lesser values from current max
82 | ( at & greater_mask ); //put in the bigger values from at
83 }
84
85 return ret;
86}

172template<typename T> std::pair<T, T> _getMinMax( const T *data, size_t len )
173{
174 size_t blocks = len / ( 16 / sizeof( T ) );
175
176 T bmin = std::numeric_limits<T>::max();
177 T bmax = std::numeric_limits<T>::min();
178
179 if( blocks ) { // if there are 16byte-blocks of values
180 std::pair<__m128i, __m128i> minmax = _getMinMaxBlockLoop<T>( reinterpret_cast<const __m128i *>( data ), blocks );
181 // compute the min/max of the blocks bmin/bmax
182 const _VectorUnion<T> smin = minmax.first;
183 const _VectorUnion<T> smax = minmax.second;
184 bmin = *std::min_element( smin.vec.elem, smin.vec.elem + 16 / sizeof( T ) );
185 bmax = *std::max_element( smax.vec.elem, smax.vec.elem + 16 / sizeof( T ) );
186 }
187
188 // if there are some remaining elements
189 if( data + blocks * 16 / sizeof( T ) < data + len ) {
190 const T rmin = *std::min_element( data + blocks * 16 / sizeof( T ), data + len );
191 const T rmax = *std::max_element( data + blocks * 16 / sizeof( T ), data + len );
192 return std::pair<T, T>( std::min( bmin, rmin ), std::max( bmax, rmax ) );
193 } else {
194 return std::pair<T, T>( bmin, bmax );
195 }
196}
