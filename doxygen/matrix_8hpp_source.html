83operator*( const util::Matrix<TYPE1, COLS, ROWS> &left, const util::Matrix<TYPE2, COLS2, COLS> &right ){
84 // transpose the right, so we can use columns as rows
85 typedef decltype(TYPE1()*TYPE2()) result_type;
86 const util::Matrix<TYPE2, COLS, COLS2> rightT = util::transpose(right);
87 util::Matrix<result_type, COLS2, ROWS> ret;
88
89 for( size_t c = 0; c < COLS2; c++ ) { //result has as many columns as right
90 const std::array<TYPE2, COLS> &rcol = rightT[c]; //columns of right are rows of rightT
91 for( size_t r = 0; r < ROWS; r++ ) { //result has as many rows as left
92 const std::array<TYPE1, COLS> &lrow = left[r];
93 ret[r][c] = std::inner_product( std::begin(lrow), std::end(lrow), std::begin(rcol), result_type() );
