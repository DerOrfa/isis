158 template<typename T> size_t insertChunksFromList ( std::list<T> &chunks, util::slist* rejected=nullptr )
159 {
160 static_assert( std::is_base_of<Chunk, T>::value, "Can only insert objects derived from Chunks" );
161 size_t cnt = 0;
162
163 for ( typename std::list<T>::iterator i = chunks.begin(); i != chunks.end(); ) { // for all remaining chunks
164 if ( insertChunk ( *i ) ) {
165 chunks.erase ( i++ );
166 cnt++;
167 } else {
168 i++;
169 }
170 }
171
172 if ( ! isEmpty() ) {
173 LOG ( Debug, info ) << "Reindexing image with " << cnt << " chunks.";
174
175 if ( !reIndex(rejected) ) {
176 LOG ( Runtime, error ) << "Failed to create image from " << cnt << " chunks.";
177 } else {
178 LOG_IF ( !getMissing().empty(), Debug, warning )
179 << "The created image is missing some properties: " << util::MSubject( getMissing() ) << ". It will be invalid.";
180 }
181 } else {
182 LOG ( Debug, warning ) << "Image is empty after inserting chunks.";
183 }
184
185 return cnt;
186 }

454 template<typename T> void copyToMem ( T *dst, size_t len) const {
455 if ( clean ) {
456 const scaling_pair scaling = getScalingTo ( util::typeName<T>() );
457
458 // we could do this using convertToType - but this solution does not need any additional temporary memory
459 for( const std::shared_ptr<Chunk> &ref : lookup ) {
460 const size_t cSize = ref->getVolume();
461
462 if ( !ref->copyToMem<T> ( dst, len, scaling ) ) {
463 LOG ( Runtime, error ) << "Failed to copy raw data of type " << ref->typeName() << " from image into memory of type " << util::typeName<T>();
464 } else {
465 if ( len < cSize ) {
466 LOG ( Runtime, error ) << "Aborting copy, because there is no space left in the target";
467 break;
468 }
469
470 len -= cSize;
471 }
472
473 dst += ref->getVolume(); // increment the cursor
474 }
475 } else {
476 LOG ( Runtime, error ) << "Cannot copy from non clean images. Run reIndex first";
477 }

779 //we want deep copies of the chunks, and we want them to be of type T
780 struct : _internal::SortedChunkList::chunkPtrOperator {

783 return std::shared_ptr<Chunk> ( new MemChunk<T> ( *ptr, scale ) );
784 }
785 } conv_op;

787 LOG ( Debug, info ) << "Computed scaling for conversion from source image: [" << conv_op.scale << "]";
788
789 this->set.transform ( conv_op );
790
791 if ( ref.isClean() ) {
792 this->lookup = this->set.getLookup(); // the lookup table still points to the old chunks
793 } else {
794 LOG ( Debug, info ) << "Copied unclean image. Running reIndex on the copy.";
795 this->reIndex();
796 }
797
