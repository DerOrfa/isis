137 template <bool CONST> static std::conditional_t<CONST,const mapped_type,mapped_type>&
138 findEntryImpl( std::conditional_t<CONST,const container_type,container_type> &root, const propPathIterator at, const propPathIterator pathEnd )
139 {
140 propPathIterator next = at;
141 next++;
142 auto found =root.find( *at );
143
144 if ( next != pathEnd ) {//we are not at the end of the path (aka the leaf)
145 if ( found != root.end() ) {//and we found the entry
146 return findEntryImpl<CONST>( found->second.branch().container, next, pathEnd ); //continue there
147 }
148 } else if ( found != root.end() ) {// if it's the leaf and we found the entry
149 return found->second; // return that entry
150 }
151
152 return nullnode();
153 }

156 template<typename T> T* queryValueAsImpl( const PropPath &path, const std::optional<size_t> &at ) {
157 const auto &found = queryProperty( path );
158
159 if( found && found->size()>at.value_or(0) ) {// apparently it has a value so let's try to use that
160 if( !found->is<T>() ) {
161 if( !found->transform<T>() ) {// convert to requested type
162 LOG( Runtime, warning ) << "Conversion of Property " << path << " from " << util::MSubject( found->getTypeName() ) << " to "
163 << util::MSubject( util::typeName<T>() ) << " failed";
164 return nullptr;
165 }
166 }
167 assert( found->is<T>() );
168 return &std::get<T>(at ?
169 found->at(*at):
170 found->front()
171 ); // use single value ops, if at was not given
172 } else
173 return nullptr;
174 }
175 template<typename T> T getValueAsImpl( const PropPath &path, const std::optional<size_t> &at )const {
176 const auto ref = queryProperty( path );
177
178 if( !ref ){
179 LOG( Runtime, warning ) << "Property " << MSubject( path ) << " doesn't exist returning " << MSubject(Value(T()) );
180 return T();
181 } else if (ref->size()<=at.value_or(0)){
182 LOG_IF(at, Runtime, warning ) << "Property " << MSubject( std::make_pair( path, ref ) ) << " does exist, but index " << *at << " is out of bounds. returning " << MSubject(Value(T()) );
183 LOG_IF(!at, Runtime, warning ) << "Property " << MSubject( std::make_pair( path, ref ) ) << " is empty. returning " << MSubject(Value(T()) );
184 return T();
185 } else
186 return at ?
187 ref->at( *at ).as<T>():
188 ref->as<T>();// use single value ops, if at was not given
189 }

493 template<typename ITER> void splice( ITER first, ITER last, bool lists_only ){
494 const PathSet empty_before=genKeyList(emptyP);
495 std::for_each( container.begin(), container.end(), Splicer<ITER>( first, last, PropPath(), lists_only) );
496 //some cleanup
497 //delete all that's empty now, but wasn't back then (we shouldn't delete what where empty before) / spliters are moved so source will become empty
498 const PathSet empty_after=genKeyList(emptyP);
499 std::list<PropPath> deletes;
500 std::set_difference(empty_after.begin(),empty_after.end(),empty_before.begin(),empty_before.end(),std::back_inserter(deletes));
501 LOG_IF(!deletes.empty(),Debug,info) << "Properties " << MSubject(deletes) << " became empty while splicing, deleting them";
502 for(const PropPath &del:deletes)
503 remove(del);
504 }

735template<typename ITER> struct PropertyMap::Splicer {
736 const ITER &first, &last;
737 const PropPath &name;
738 const bool lists_only;
739 const size_t blocks;
740 Splicer( const ITER &_first, const ITER &_last, const PropPath &_name, bool _lists_only ):
741 first( _first ), last( _last ), name( _name ),lists_only(_lists_only), blocks(std::distance( first, last ))
742 {
743 assert( blocks );
744 }
745 void operator()( container_type::value_type &pair )const {
746 const auto &key=pair.first;
747 Node &node= pair.second;
748 std::visit( Splicer<ITER>( first, last, name / key, lists_only ), node.variant() );
749 }
750 void operator()( const std::monostate &val )const {}
751 void operator()( PropertyValue &val )const {
752 if(val.isEmpty())return; // abort if there is nothing to spliceAt
753 else if(lists_only && val.size() <= 1)return; //abort if we don't want to move scalars
754
755 if( val.size() % blocks ) { // just copy all which cannot be properly spliced to the destination
756 if(val.size() > 1){
757 LOG_IF( val.size() > 1, Runtime, verbose_info ) << "Not splicing non scalar property " << MSubject( name ) << " because its length "
758 << MSubject( val.size() ) << " doesn't fit the amount of targets(" << MSubject( blocks ) << ")"; //tell the user if its no scalar
759 }
760 PropertyValue &first_prop = _internal::un_shared_ptr(*first).touchProperty( name );
761 first_prop.transfer(val);
762 ITER i = first;
763 for( ++i; i != last; ++i )
764 _internal::un_shared_ptr(*i).touchProperty( name ) = first_prop;
765
766 } else {
767 LOG_IF( val.size() > 1, Debug, info ) << "Splicing non scalar property " << MSubject( name ) << " into " << blocks << " chunks";
768 ITER i = first;
769 for( PropertyValue & splint : val.splice( val.size() / blocks ) ) {
770 assert( i != last );
771 _internal::un_shared_ptr(*i).touchProperty( name ).swap(splint);
772 i++;
773 }
774 }
775 assert(val.isEmpty());
776 }
777 void operator()( PropertyMap &sub )const { //call my own recursion for each element
778 std::for_each( sub.container.begin(), sub.container.end(), *this );
779 }
