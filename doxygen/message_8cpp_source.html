127std::string Message::merge(const std::string& color_code)const
128{
129 const std::string reset_code(color_code.empty()? "":"\033[0m");
130 const std::string s_prefix(color_code.empty()? "\"":"\x1B[1m");
131 const std::string s_suffix(color_code.empty()? std::string("\""):reset_code+color_code);
132
133 std::string ret( str() );
134 size_t found = std::string::npos;
135 auto subj = m_subjects.begin();
136
137 if ( ( found = ret.find( "{o}" ) ) != std::string::npos ){
138 ret.replace( found, 3, m_object );
139 found+=m_object.length();
140 }
141
142 found = 0;
143
144 while ( ( found = ret.find( "{s}", found ) ) != std::string::npos ){
145 std::string insert=s_prefix + * ( subj++ ) + s_suffix;
146 ret.replace( found, 3, insert );
147 found+=insert.length();
148 }
149
150 return color_code+ret+reset_code;
151}

188void DefaultMsgPrint::commit_tty(const Message& mesg)
189{
190 const char *color_code="";
191
192 static int erret = 0;
193 static bool is_term =
194#ifdef HAVE_CURSES
195 (setupterm(nullptr, 1, &erret) == 0);
196#else
197 false;
198#endif
199
200 // terminal color codes
201 static const char
202 red_code[]="\x1B[31m",
203 yellow_code[]="\x1B[33m",
204 green_code[]="\x1B[32m",
205 white_code[]="\x1B[37m",
206 norm_code[]="\x1B[0m";
207
208 if(is_term)
209 switch(mesg.m_level){
210 case error:color_code=red_code;break;
211 case warning:color_code=yellow_code;break;
212 case notice:color_code=green_code;break;
213 case info:color_code=white_code;break;
214 default:color_code=norm_code;break;
215 }
216
217#ifndef NDEBUG //if with debug-info
218 fprintf(
219 stderr,"\r%s:%s[%s:%d]%s\n",
220 mesg.m_module.c_str(),
221 logLevelName( mesg.m_level ),
222 mesg.m_file.filename().c_str(),
223 mesg.m_line,
224 mesg.merge(color_code).c_str()
225 );
226#else
227 fprintf(
228 stderr,"\r%s:%s[%s]%s\n",
229 mesg.m_module.c_str(),
230 logLevelName( mesg.m_level ),
231 mesg.m_object.c_str(),
232 mesg.merge(color_code).c_str()
233 );
234#endif //NDEBUG
235}
