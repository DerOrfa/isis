172std::list<Chunk> Chunk::autoSplice ( dimensions atDim )const
173{
174 if ( !PropertyMap::isValid() ) {
175 LOG( Runtime, error ) << "Cannot spliceAt invalid Chunk (missing properties are " << this->getMissing() << ")";
176 return {};
177 }
178
179 const auto distance = getValueAs<util::fvector3>( "voxelSize" ) + getValueAsOr( "voxelGap",util::fvector3());
180
181 LOG_IF( atDim < data::timeDim && distance[atDim] == 0, Runtime, error ) << "The voxel distance (voxelSize + voxelGap) at the splicing direction (" << atDim << ") is zero. This will likely cause errors in the Images structure.";
182
183 const auto orgsize=getSizeAsVector();
184
185 util::PropertyMap spliceMap = *this;
186 auto &indexOrigin = spliceMap.touchProperty("indexOrigin");
187 auto &acquisitionNumber = spliceMap.touchProperty("acquisitionNumber");
188 if(indexOrigin.isEmpty()){
189 indexOrigin = util::dvector3{0,0,0};
190 LOG(Runtime,error) << "Trying to splice a chunk without indexOrigin. Assuming " << indexOrigin;
191 }
192 if(acquisitionNumber.isEmpty()){
193 acquisitionNumber = 0u;
194 LOG(Runtime,error) << "Trying to splice a chunk without acquisitionNumber. Assuming " << acquisitionNumber;
195 }
196
197 std::vector vecs{
198 getValueAs<util::dvector3>("rowVec"),
199 getValueAs<util::dvector3>("columnVec")
200 };
201 if(auto svec=queryProperty( "sliceVec" ); svec ) {
202 vecs.push_back(svec->as<util::dvector3>());
203 } else { // if sliceVec isn't given, compute it as Normal on rowVec and columnVec
204 assert( util::fuzzyEqual<float>( vecs[0].sqlen(), 1 ) );
205 assert( util::fuzzyEqual<float>( vecs[1].sqlen(), 1 ) );
206 vecs.push_back({
207 vecs[0][1] * vecs[1][2] - vecs[0][2] * vecs[1][1],
208 vecs[0][2] * vecs[1][0] - vecs[0][0] * vecs[1][2],
209 vecs[0][0] * vecs[1][1] - vecs[0][1] * vecs[1][0]
210 });
211 }
212
213 // go through each to-splice dimension and explode indexOrigin and acquisitionNumber with the size of the given
214 // dimension if they aren't already
215 size_t stepsize=1;
216 for(auto dim=getRelevantDims()-1;dim>=atDim;--dim){
217 stepsize*=orgsize[dim];//expected size of the property grow with every processed dimension by this dimension
218 if(indexOrigin.size()<stepsize){
219 LOG_IF(stepsize%indexOrigin.size(),Runtime,error)
220 << "The splicing stepsize " << stepsize
221 << " is not a multiple of the amount of values in indexOrigin ("
222 << indexOrigin.size() << "). This splice is most likely going to fail.";
223 if(dim<timeDim){
224 auto offset = vecs[dim] * distance[dim]; //direction and length to the next splice
225 auto op = [offset,dimsize=orgsize[dim], cnt=0](const util::Value &v)mutable->util::Value{
226 return v.as<util::dvector3>() + offset * (cnt++ % dimsize);
227 };
228 LOG(Debug,verbose_info) << "Stretching indexOrigin " << indexOrigin << " by " << orgsize[dim];
229 indexOrigin.explode(orgsize[dim],op);
230 } else // on timeDim indexOrigin is constant
231 indexOrigin.explode(orgsize[dim],[](const util::Value &v){return v;});
232 assert(indexOrigin.size()==stepsize);
233 }
234 if(acquisitionNumber.size()<stepsize){
235 LOG_IF(stepsize%acquisitionNumber.size(),Runtime,error)
236 << "The splicing stepsize " << stepsize
237 << " is not a multiple of the amount of values in acquisitionNumber ("
238 << acquisitionNumber.size() << "). This splice is most likely going to fail.";
239 auto op = [dimsize=orgsize[dim], cnt=0](const util::Value &v)mutable->util::Value{
240 const auto iv = v.as<uint32_t>();
241 return {uint64_t(iv * dimsize + (cnt++ % dimsize))};
242 };
243 LOG(Debug,verbose_info) << "Stretching acquisitionNumber " << acquisitionNumber << " by " << orgsize[dim];
244 acquisitionNumber.explode(orgsize[dim],op);
245 assert(acquisitionNumber.size()==stepsize);
246 }
247 }
248
249 LOG(Debug, info ) << "Splicing chunk at dimension " << atDim + 1;
250 LOG(Debug, info) << "indexOrigin-list is " << indexOrigin;
251 LOG(Debug,info) << "acquisitionNumber list is " << acquisitionNumber;
252
253 // do the actual splicing with the modified PropertyMap
254 return spliceAt(atDim,std::move(spliceMap));
255}

260std::list<Chunk> Chunk::spliceAt (dimensions atDim, util::PropertyMap &&propSource )const
261{
262 std::list<Chunk> ret;
263
264 //@todo should be locking
265 const util::vector<size_t, dims> wholesize = getSizeAsVector();
266 util::vector<size_t, dims> spliceSize{1,1,1,1};
267 //copy the relevant dimensional sizes from wholesize (in case of sliceDim we copy only the first two elements of wholesize - making slices)
268 std::copy(std::begin(wholesize),std::begin(wholesize)+atDim,std::begin(spliceSize));
269 //get the spliced ValueArray's (the volume of the requested dims is the split-size - in case of sliceDim it is rows*columns)
270 const auto pointers = ValueArray::splice(spliceSize.product() );
271 assert(pointers.size()==wholesize.product()/spliceSize.product());
272
273 //create new Chunks from this ValueArray's
274 for( const ValueArray &ref : pointers ) {
275 ret.push_back( Chunk( ref, spliceSize[0], spliceSize[1], spliceSize[2], spliceSize[3] ) );
276 }
277 propSource.splice(ret.begin(),ret.end(),false);//copy/spliceAt properties into spliced chunks / this will empty propSource
278 return ret;
279}

281void Chunk::flipAlong( const dimensions dim )
282{
283 const size_t elSize = getBytesPerVoxel();
284 const util::vector4<size_t> whole_size = getSizeAsVector();
285
286 auto swap_ptr = std::static_pointer_cast<uint8_t>( getRawAddress() );
287 uint8_t *swap_start = swap_ptr.get();
288 const uint8_t *const swap_end = swap_start + whole_size.product() * elSize;
289
290 size_t block_volume = whole_size.product();
291
292 for( int i = data::timeDim; i >= dim; i-- ) {
293 assert( ( block_volume % whole_size[i] ) == 0 );
294 block_volume /= whole_size[i];
295 }
296
297 assert( block_volume );
298 block_volume *= elSize;
299 const size_t swap_volume = block_volume * whole_size[dim];
300 uint8_t buff[ block_volume ];
301
302 //iterate over all swap-volumes
303 for( ; swap_start < swap_end; swap_start += swap_volume ) { //outer loop
304 // swap each block with the one at the oppsite end of the swap_volume
305 uint8_t *a = swap_start; //first block
306 uint8_t *b = swap_start + swap_volume - block_volume; //last block within the swap-volume
307
308 for( ; a < b; a += block_volume, b -= block_volume ) { // grow a, shrink b (inner loop)
309 memcpy( buff, a, block_volume );
310 memcpy( a, b, block_volume );
311 memcpy( b, buff, block_volume );
312 }
313
314 }
