50template<typename TARGET, typename traits>
51bool stringTo(const std::basic_string<char, traits> &str, TARGET& target){
52 if constexpr(std::is_integral_v<TARGET> && !std::is_same_v<bool,TARGET>){
53 auto [p,ec] = std::from_chars(str.data(),str.data()+str.size(),target);
54 return ec == std::errc();
55 } else if constexpr(std::is_floating_point_v<TARGET>){
56 return stringTofloat(str,target);// @todo get rid of this once from_chars is available for float
57 } else if constexpr(std::is_same_v<std::string,TARGET>){
58 target=std::string(str.data(),str.length());
59 return true;
60 } else if constexpr(std::is_same_v<istring,TARGET>){
61 target=istring(str.data(),str.length());
62 return true;
63 } else { //
64 std::stringstream stream({str.data(),str.length()});
65 stream>>target;
66 bool eof = stream.eof() || stream.tellg()>=str.length();
67 LOG_IF(stream.fail(),CoreDebug,error) << "Failed to read string " << str;
68 LOG_IF(!eof,CoreDebug,warning)
69 << "Didn't use all " << str << " while converting it to " << target << " stopped at " << str.substr(stream.tellg());
70 return !stream.fail() && eof;
71 }
72}
