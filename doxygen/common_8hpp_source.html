52template<typename ForwardIterator, typename T, typename CMP>
53bool
54continousFind(ForwardIterator &current, const ForwardIterator end, const T &compare, const CMP &compOp)
55{
56 //find the first iterator which does not compare less
57 current = std::lower_bound(current, end, compare, compOp);
58
59 if (current == end //if we're at the end
60 || compOp(compare, *current) //or compare less than that iterator (e.g. iterator greater than compare)
61 )
62 return false;//we didn't find a match
63 else
64 return true;//not(current <> compare) makes compare == current
65}

134template<typename T>
135bool fuzzyEqual(T a, T b, unsigned short scale = 10)
136{
137 static_assert(std::is_floating_point<T>::value, "must be called with floating point");
138
139 static const T
140 epsilon = std::numeric_limits<T>::epsilon(); // get the distance between 1 and the next representable value
141 T bigger, smaller;
142
143 a = std::abs(a);
144 b = std::abs(b);
145
146 if (a < b) {
147 bigger = b;
148 smaller = a;
149 }
150 else {
151 smaller = b;
152 bigger = a;
153 }
154
155 if (smaller == 0)
156 return bigger < (epsilon * scale);
157
158 const T factor = 1 / smaller; // scale smaller to that value
159 return (bigger * factor)
160 <= (1 + epsilon * scale); //scaled bigger should be between 1 and the next representable value
161}

265template<typename charT, typename _Tp, typename _Compare, typename _Alloc >
266basic_ostream<charT, std::char_traits<charT> >& operator<<( basic_ostream<charT, std::char_traits<charT> > &out, const map<std::string, _Tp, _Compare, _Alloc>& s )
267{
268 size_t key_len = 0;
269 typedef typename map<std::string, _Tp, _Compare, _Alloc>::const_iterator m_iterator;
270
271 for ( m_iterator i = s.begin(); i != s.end(); i++ )
272 if ( key_len < i->first.length() )
273 key_len = i->first.length();
274
275 for ( m_iterator i = s.begin(); i != s.end(); ) {
276 out << make_pair( i->first + std::string( key_len - i->first.length(), ' ' ), i->second );
277
278 if ( ++i != s.end() )
279 out << std::endl;
280 }
281
282 return out;
283}
