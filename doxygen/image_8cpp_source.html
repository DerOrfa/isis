94void Image::swapDim( short unsigned int dim_a, short unsigned int dim_b, std::shared_ptr<util::ProgressFeedback> feedback )
95{
96 NDimensional<4>::swapDim( dim_a, dim_b, begin(), feedback ); // this runs through all chunks as the Iterator does that
97 for(auto pCh:lookup){ //but we still have to reshape the chunks
98 auto shape=pCh->getSizeAsVector();
99 std::swap(shape[dim_a],shape[dim_b]);
100 pCh->init(shape);
101 }
102
103 //swap voxel sizes
104 auto voxel_size_query=queryValueAs<util::fvector3>("voxelSize");
105 if(voxel_size_query && dim_a<data::timeDim && dim_b<data::timeDim){
106 util::fvector3 &voxel_size=*voxel_size_query;
107 std::swap(voxel_size[dim_a],voxel_size[dim_b]);
108 }
109
110 //voxelsize is needed to be equal inside Images so there should be no voxelSize in the chunks
111 assert(!getChunkAt(0,false).hasProperty("voxelSize"));
112}

115void Image::deduplicateProperties()
116{
117
118 LOG_IF( lookup.empty(), Debug, error ) << "The lookup table is empty. Won't do anything.";
119 if(lookup.empty())
120 return;
121
122 //transform lookup into a "maps-list"
123 std::list<std::shared_ptr<PropertyMap>> maps;
124 std::transform(
125 lookup.begin(),lookup.end(),std::back_inserter(maps),
126 [](const std::shared_ptr<Chunk> &m){
127 return std::static_pointer_cast<PropertyMap>(m);
128 }
129 );
130
131 PropertyMap::deduplicate(maps); // call PropertyMap's deduplicate
132 maps.clear();
133
134 // if there are some non-spliced..
135 std::list<std::list<std::shared_ptr<Chunk>>> chunks;
136 if(!set.not_spliced.empty()){ //deduplicate them as well
137
138 // first get them into the proper container
139 for(auto &pair:set.not_spliced){
140 maps.emplace_back(new util::PropertyMap());
141 maps.back()->transfer(pair.first);
142 chunks.push_back({});
143 chunks.back().swap(pair.second);
144 }
145 set.not_spliced.clear();
146 PropertyMap::deduplicate(maps); // call PropertyMap's deduplicate on the not_spliced
147
148 // and copy into the chunks what's left
149 auto c=chunks.begin();
150 for(auto m=maps.begin();m!=maps.end();m++,c++){
151 LOG(Debug,verbose_info) << "Copying true splices " << **m << " into " << c->size() << " chunks";
152 for(auto &ch:*c){
153 ch->join(**m);
154 }
155 }
156 }
157
158 LOG( Debug, verbose_info ) << "common properties now in the image: " << *this;
159}

161bool Image::insertChunk ( const Chunk &chunk )
162{
163 if ( chunk.getVolume() == 0 ) {
164 LOG( Runtime, error )
165 << "Cannot insert empty Chunk (Size is " << chunk.getSizeAsString() << ").";
166 return false;
167 }
168
169 if ( !chunk.isValid() ) {
170 LOG( Runtime, error ) << "Cannot insert invalid Chunk.";
171 return false;
172 }
173
174 if( clean ) {
175 LOG( Runtime, warning ) << "Inserting into already indexed images is inefficient. You should not do that.";
176
177 // re-move all properties from the image back into the chunks
178 for( std::shared_ptr<Chunk> &ref : lookup ) {
179 ref->transfer( *this );
180 }
181 }
182
183
184 if( set.insert( chunk ) ) { // if the insertion was successful the image has to be reindexed anyway
185 clean = false;
186 lookup.clear();
187 return true;
188 } else {
189 // if the insertion failed, but the image was clean (aka we above moved properties back into chunks)
190 // so we have to de-duplicate properties again
191 // the image is still clean though, no need to reindex
192 if( clean )
193 deduplicateProperties();
194
195 return false;
196 }

209bool Image::reIndex(util::slist* rejected)
210{
211 if ( set.isEmpty() ) {
212 LOG( Debug, warning ) << "Reindexing an empty image is useless.";
213 return false;
214 }
215
216 if( set.makeRectangular(rejected) > 0 && set.isEmpty() ) {
217 LOG( Runtime, error ) << "No retangular image data left. Skipping";
218 return false;
219 }
220
221 //redo lookup table
222 lookup = set.getLookup();
223 util::vector<size_t, dims> structure_size; //storage for the size of the chunk structure
224 structure_size.fill( 1 );
225
226 //get primary attributes from geometrically first chunk - will be useful
227 const Chunk &first = chunkAt( 0 );
228
229 //start indexing at either the chunk-size or the give minIndexingDim (whichever is bigger)
230 const unsigned short chunk_dims = std::max<unsigned short>( first.getRelevantDims(), minIndexingDim );
231 chunkVolume = first.getVolume();
232
233 //move all props from the image and move them (back) into the chunks to redo common check
234 for(size_t i=0;i+1<lookup.size();i++) // copy the first n-1
235 lookup[i]->join(*this);
236 lookup.back()->transfer(*this); //transfer into the last
237
238 //clean generated props if they're still here (which would mean the chunks has their own -- which is a good thing)
239 if(hasProperty("indexOrigin"))remove( "indexOrigin" );
240 if(hasProperty("rowVec"))remove( "rowVec" );
241 if(hasProperty("columnVec"))remove( "columnVec" );
242 if(hasProperty("sliceVec"))remove( "sliceVec" );
243
245 // Determine structure of the image by searching for dimensional breaks in the chunklist
247
248 //if there are many chunks, they must leave at least on dimension to the image to "sort" them in
249 const size_t timesteps = set.getHorizontalSize();
250 const unsigned short sortDims = dims - ( timesteps > 1 ? 1 : 0 ); // don't use the uppermost dim, if the timesteps are already there
251
252 if ( chunk_dims >= Image::dims ) {
253 if ( lookup.size() > 1 ) {
254 LOG( Runtime, error )
255 << "Cannot handle multiple Chunks, if they have more than "
256 << Image::dims - 1 << " dimensions";
257 return false;
258 }
259
260 //if there is only one chunk, it's ok - the image will consist only of this one,
261 //and commonGet will always return <0,set.begin()->getLinearIndex()>
262 //thus in this case voxel() equals set.begin()->voxel()
263 } else {// OK there is at least one dimension to sort in the chunks
264 LOG( Debug, info ) << "Computing strides for dimensions " << util::MSubject( chunk_dims + 1 ) << " to " << util::MSubject( sortDims );
265
266 // check the chunks for at least one dimensional break - use that for the size of that dimension
267 for ( unsigned short i = chunk_dims; i < sortDims; i++ ) { //if there are dimensions left figure out their size
268 structure_size[i] = getChunkStride( structure_size.product() ) / structure_size.product();
269 assert( structure_size[i] != 0 );
270 }
271 }
272
273 if ( sortDims < dims ) { //if there is a timedim (not all dims was used for geometric sort)
274 assert( structure_size[sortDims] == 1 );
275 structure_size[sortDims] = timesteps; // fill the dim above the top geometric dim with the timesteps
276 }
277
278 assert( structure_size.product() == lookup.size() );
279 //Clean up the properties
280 deduplicateProperties();
281
282 // add the chunk-size to the image-size
283 for ( unsigned short i = 0; i < chunk_dims; i++ )
284 structure_size[i] = first.getDimSize( i );
285
286 init( structure_size ); // set size of the image
288 //reconstruct some redundant information, if its missing
290 const util::PropertyMap::key_type vectors[] = {"rowVec", "columnVec", "sliceVec"};
291 int oneCnt = 0;
292
293 for( const util::PropertyMap::key_type & ref : vectors ) {
294 const auto found=queryValueAs<util::fvector3>( ref );
295 if ( found ) {
296 util::fvector3 &vec = *found;
297
298 if( vec.sqlen() == 0 ) {
299 util::fvector3 v_one{0,0,0};
300 v_one[oneCnt] = 1;
301 LOG( Runtime, error )
302 << "The existing " << ref << " " << vec << ( hasProperty( "source" ) ? " from " + getValueAs<std::string>( "source" ) : "" )
303 << " has the length zero. Falling back to " << v_one << ".";
304 vec = v_one;
305 }
306
307 vec.normalize();
308 }
309 oneCnt++;
310 }
311
312 if(!hasProperty("indexOrigin")){ // if there is no common indexOrigin
313 const util::PropertyValue* found =first.queryProperty("indexOrigin");
314 if(found) // get it from the first chunk - which than by definition should have one
315 touchProperty("indexOrigin")=*found;
316 else{
317 LOG(Runtime,error) << "No indexOrigin found " << " falling back to " << util::fvector3();
318 setValueAs("indexOrigin",util::fvector3());
319 }
320 }
321
322 // check voxelsize
323 util::fvector3 &voxeSize = refValueAs<util::fvector3>( "voxelSize" );
324 for( int i = 0; i < 3; i++ ) {
325 if( voxeSize[i] == 0 || std::isinf( voxeSize[i] ) ) {
326 LOG( Runtime, warning ) << "voxelSize[" << i << "]==" << voxeSize[i] << " is invalid, using 1";
327 voxeSize[i] = 1;
328 }
329 }
330
331
332 //if we have at least two slides (and have slides (with different positions) at all)
333 const util::PropertyValue* firstV = first.queryProperty( "indexOrigin" );
334
335 if ( chunk_dims == 2 && structure_size[2] > 1 && firstV ) {
336 const Chunk &last = chunkAt( structure_size[2] - 1 );
337
338 const util::PropertyValue* lastV = last.queryProperty( "indexOrigin" );
339
340 if ( lastV ) {
341 //check the slice vector
342 util::fvector3 distVecNorm = lastV->as<util::fvector3>() - firstV->as<util::fvector3>();
343 LOG_IF( distVecNorm.sqlen() == 0, Runtime, error )
344 << "The distance between the the first and the last chunk is zero. That's bad, because I'm going to normalize it.";
345 distVecNorm.normalize();
346
347 const auto sliceVec = queryValueAs<util::fvector3>( "sliceVec" );
348
349 if ( sliceVec ) {
350 LOG_IF( ! distVecNorm.fuzzyEqual(*sliceVec) , Runtime, info )
351 << "The existing sliceVec " << *sliceVec
352 << " differs from the distance vector between chunk 0 and " << structure_size[2] - 1
353 << " " << distVecNorm;
354 } else {
355 LOG( Debug, info )
356 << "used the distance between chunk " << 0 << " and " << structure_size[2] - 1
357 << " to synthesize the missing sliceVec as " << distVecNorm;
358 setValueAs( "sliceVec", distVecNorm); // this should message if there really is a conflict
359 }
360
361 const float avDist = ( lastV->as<util::fvector3>() - firstV->as<util::fvector3>()).len() / ( structure_size[2] - 1 ); //average dist between the middle of two slices
362
363 const float sliceDist = avDist - voxeSize[2]; // the gap between two slices
364
365 if ( sliceDist > 0 ) {
366 static const float inf = -std::numeric_limits<float>::infinity();
367
368 util::fvector3 &voxelGap = refValueAsOr( "voxelGap", util::fvector3{0, 0, inf} ); //if there is no voxelGap yet, we create it as (0,0,inf)
369
370 if ( voxelGap[2] != inf ) {
371 if(! util::fuzzyEqual( voxelGap[2], sliceDist, 500 )){
372 LOG(Runtime, warning )
373 << "The existing slice distance (voxelGap[2]) " << util::MSubject( voxelGap[2] )
374 << " differs from the distance between chunk 0 and 1, which is " << util::MSubject( sliceDist );
375 }
376 } else {
377 voxelGap[2] = sliceDist;
378 LOG( Debug, info )
379 << "used the distance between chunk " << 0 << " and " << 1 << " to synthesize the missing slice distance (voxelGap[2]) as "
380 << util::MSubject( sliceDist );
381 }
382 }
383 }
384 }
385
386 //if we have row- and column- vector
387 auto rrow = queryValueAs<util::fvector3>( "rowVec" ),rcolumn = queryValueAs<util::fvector3>( "columnVec" );
388 if(rrow && rcolumn){
389 const util::fvector3 &row=*rrow,&column=*rcolumn;
390 LOG_IF( row.dot( column ) > 0.01f, Runtime, warning ) << "The cosine between the columns and the rows of the image is bigger than 0.01";
391 const util::fvector3 crossVec = util::fvector3({ //we could use their cross-product as sliceVector
392 row[1] * column[2] - row[2] * column[1],
393 row[2] * column[0] - row[0] * column[2],
394 row[0] * column[1] - row[1] * column[0]
395 });
396 auto sliceVec = queryValueAs<util::fvector3>( "sliceVec" );
397
398 if ( sliceVec ) {
399 LOG_IF( std::acos( crossVec.dot(*sliceVec) ) > 180 / M_PI, Runtime, warning ) //angle more than one degree
400 << "The existing sliceVec " << util::MSubject( sliceVec ) << " differs from the cross product of the row- and column vector " << util::MSubject( crossVec );
401 } else {
402 // We don't know anything about the slice-direction
403 // we just guess its along the positive cross-product between row- and column direction
404 // so at least warn the user if we do that long shot
405 LOG( Runtime, info )
406 << "used the cross product between rowVec and columnVec as sliceVec:"
407 << crossVec << ". That might be wrong!";
408 setValueAs( "sliceVec", crossVec );
409 }
410 } else {
411 LOG(Runtime,warning) << "Image has no common rowVec/columnVec, won't check for sliceVec";
412 }
413
414
415 auto storedFoV = queryValueAs<util::fvector3>( "fov" );
416
417 if ( storedFoV ) {
418 const util::fvector3 calcFoV = getFoV();
419
420 bool ok = true;
421
422 for ( size_t i = 0; i < 3; i++ ) {
423 if ( ( *storedFoV )[i] != -std::numeric_limits<float>::infinity() ) {
424 ok &= util::fuzzyEqual( ( *storedFoV )[i], calcFoV[i] );
425 } else
426 ( *storedFoV )[i] = calcFoV[i];
427 }
428
429 LOG_IF( ! ok, Runtime, info )
430 << "The calculated field of view differs from the stored " << util::MSubject( *storedFoV ) << "/" << calcFoV;
431 }
432
433 LOG_IF( ! isValid(), Runtime, warning ) << "The image is not valid after reindexing. Missing properties: " << getMissing();
434
435 return clean = isValid();
436}

476void Image::copyToValueArray(ValueArray &dst, scaling_pair scaling ) const
477{
478 if( getVolume() > dst.getLength() ) {
479 LOG( Runtime, error ) << "Image won't fit into the ValueArray, wont copy..";
480 return;
481 }
482
483 if ( clean ) {
484 if(!scaling.valid)
485 scaling = getScalingTo ( dst.getTypeID() );
486
487 assert(scaling.valid);
488
489 std::vector<ValueArray > targets;
490
491 if( lookup.size() > 1 ) { //if there are more than 1 chunks
492 //splice target to have the same parts as the image
493 targets = dst.splice( lookup.front()->getVolume() );
494 } else {
495 //just put that ValueArray into the list
496 targets.push_back( dst );
497 }
498 assert(targets.size()==lookup.size());
499
500 auto target = targets.begin();
501 for( const std::shared_ptr<Chunk> &ref : lookup ) { // copy chunks into the parts
502 if ( !ref->copyTo ( *target, scaling ) ) {
503 LOG ( Runtime, error )
504 << "Failed to copy raw data of type " << ref->typeName() << " from " << getSizeAsString() << "-image into ValueArray of type "
505 << dst.typeName() << " and length " << dst.getLength();
506 }
507
508 target++;
509 }
510 } else {
511 LOG ( Runtime, error ) << "Cannot copy from non clean images. Run reIndex first";
512 }
513}

516 Image ret( *this ); // ok we just cheap-copied the whole image
517
518 //we want deep copies of the chunks, and we want them to be of type ID
519 struct : _internal::SortedChunkList::chunkPtrOperator {

521 return std::make_shared<Chunk> ( ptr->copyByID( ptr->getTypeID(), scaling_pair(1,0) ) );
522 }
523 } conv_op;
524
525 ret.set.transform ( conv_op );
526
527 if ( ret.isClean() ) {
528 ret.lookup = ret.set.getLookup(); // the lookup table still points to the old chunks
529 } else {
530 LOG ( Debug, info ) << "Copied unclean image. Running reIndex on the copy.";
531 ret.reIndex();
532 }
533
534 return ret;

582size_t Image::getChunkStride ( size_t base_stride )
583{
584 LOG_IF( set.isEmpty(), Runtime, error ) << "Trying to get chunk stride in an empty image";
585 LOG_IF( lookup.empty(), Debug, error ) << "Lookup table for chunks is empty. Do reIndex() first!";
586
587 if ( lookup.size() >= 4 * base_stride ) {
588 /* there can't be any stride with less than 3*base_stride chunks (which would actually be an invalid image)
589 * _____
590 * |c c| has no stride/dimensional break
591 * _____
592 * |c c|
593 * |c | has a dimensional break, but is invalid
594 * _____
595 * |c c|
596 * |c c| is the first reasonable case
597 */
598 // get the distance between first and second chunk for comparision
599 const util::fvector3 firstV = chunkAt( 0 ).getValueAs<util::fvector3>( "indexOrigin" );
600 const util::fvector3 secondV = chunkAt( base_stride ).getValueAs<util::fvector3>( "indexOrigin" );
601 const util::fvector3 dist1 = secondV - firstV;
602
603 if( dist1.sqlen() == 0 ) { //if there is no geometric structure anymore - so asume its flat from here on
604 LOG( Debug, info ) << "Distance between 0 and " << util::MSubject( base_stride )
605 << " is zero. Assuming there are no dimensional breaks anymore. Returning " << util::MSubject( base_stride );
606 return base_stride;
607 } else for ( size_t i = base_stride; i < lookup.size() - base_stride; i += base_stride ) { // compare every follwing distance to that
608 const util::fvector3 thisV = chunkAt( i ).getValueAs<util::fvector3>( "indexOrigin" );
609 const util::fvector3 nextV = chunkAt( i + base_stride ).getValueAs<util::fvector3>( "indexOrigin" );
610 const util::fvector3 distFirst = nextV - firstV;
611 const util::fvector3 distThis = nextV - thisV;
612 LOG( Debug, verbose_info )
613 << "Distance between chunk " << util::MSubject( i ) << " and " << util::MSubject( i + base_stride )
614 << " is " << distThis.len() << ". Distance between 0 and " << util::MSubject( i + base_stride ) << " is " << distThis.len();
615
616 if ( distFirst.sqlen() <= distThis.sqlen() ) { // the next chunk is nearer to the begin than to this => dimensional break => leave
617 LOG( Debug, info )
618 << "Distance between chunk " << util::MSubject( i + base_stride )
619 << " and 0 is not bigger than the distance between " << util::MSubject( i + base_stride )
620 << " and " << util::MSubject( i ) << ", assuming dimensional break at " << i + base_stride;
621 return i + base_stride;
622 }
623 }
624 } else if ( lookup.size() % base_stride ) {
625 LOG( Runtime, error )
626 << "The amount of chunks (" << lookup.size()
627 << ") is not divisible by the block size of the dimension below (" << base_stride
628 << "). Maybe the image is incomplete.";
629 LOG( Runtime, warning )
630 << "Ignoring " << lookup.size() % base_stride << " chunks.";
631 return lookup.size() - ( lookup.size() % base_stride );
632 }

720size_t Image::compare( const isis::data::Image &comp ) const
721{
722 size_t ret = 0;
723 LOG_IF( ! ( clean && comp.clean ), Debug, error )
724 << "Comparing unindexed images will cause you trouble, run reIndex()!";
725
726 if ( getSizeAsVector() != comp.getSizeAsVector() ) {
727 LOG( Runtime, warning ) << "Size of images differs (" << getSizeAsVector() << "/"
728 << comp.getSizeAsVector() << "). Adding difference to the result.";
729 ret += ( getSizeAsVector() - comp.getSizeAsVector() ).product();
730 }
731
732 const util::vector4<size_t> firstVec=chunkPtrAt( 0 )->getSizeAsVector(),secondVec=comp.chunkPtrAt( 0 )->getSizeAsVector();
733 util::vector4<size_t> minVector;
734 std::transform(
735 std::begin(firstVec),std::end(firstVec),
736 std::begin(secondVec),std::begin(minVector),
737 [](size_t first,size_t second){return std::min(first,second);}
738 );
739 const size_t increment = minVector.product();
740
741 for ( size_t i = 0; i < getVolume(); i += increment ) {
742 const size_t nexti = i + increment - 1;
743 const std::pair<size_t, size_t> c1pair1( i / chunkVolume, i % chunkVolume );
744 const std::pair<size_t, size_t> c1pair2( nexti / chunkVolume, nexti % chunkVolume );
745 const std::pair<size_t, size_t> c2pair1( i / comp.chunkVolume, i % comp.chunkVolume );
746 assert( c1pair1.first == c1pair2.first );
747 LOG( Debug, verbose_info ) << "Comparing chunks at " << c1pair1.first << " and " << c2pair1.first;
748 const Chunk &c1 = *chunkPtrAt( c1pair1.first );
749 const Chunk &c2 = *( comp.chunkPtrAt( c2pair1.first ) );
750 LOG( Debug, verbose_info )
751 << "Start positions are " << c1pair1.second << " and " << c2pair1.second
752 << " and the length is " << c1pair2.second - c1pair1.second;
753 ret += static_cast<const ValueArray&>(c1).compare(c1pair1.second, c1pair2.second, c2, c2pair1.second );
754 }
755
756 return ret;
757}

759Image::orientation Image::getMainOrientation()const
760{
761 LOG_IF( ! isValid() || ! clean, Debug, warning ) << "You should not run this on non clean image. Run reIndex first.";
762 auto row = getValueAs<util::fvector3>( "rowVec" );
763 auto column = getValueAs<util::fvector3>( "columnVec" );
764 row.normalize();
765 column.normalize();
766 LOG_IF( row.dot( column ) > 0.01, Runtime, warning ) << "The cosine between the columns and the rows of the image is bigger than 0.01";
767 const util::fvector3 crossVec = util::fvector3({
768 row[1] * column[2] - row[2] * column[1],
769 row[2] * column[0] - row[0] * column[2],
770 row[0] * column[1] - row[1] * column[0]
771 });
772 const util::fvector3 x({1, 0}), y({0, 1}), z({0, 0, 1});
773 double a_axial = std::acos( crossVec.dot( z ) ) / M_PI;
774 double a_sagittal = std::acos( crossVec.dot( x ) ) / M_PI;
775 double a_coronal = std::acos( crossVec.dot( y ) ) / M_PI;
776 bool a_inverse = false, s_inverse = false, c_inverse = false;
777 LOG( Debug, info ) << "Angles to vectors are " << ( a_sagittal * 180 ) << " to x, " << ( a_coronal * 180 ) << " to y and " << ( a_axial * 180 ) << " to z";
778
779 if( a_axial > .5 ) {
780 a_axial = std::abs( a_axial - 1 );
781 a_inverse = true;
782 }
783
784 if( a_sagittal > .5 ) {
785 a_sagittal = std::abs( a_sagittal - 1 );
786 s_inverse = true;
787 }
788
789 if( a_coronal > .5 ) {
790 a_coronal = std::abs( a_coronal - 1 );
791 c_inverse = true;
792 }
793
794 if( a_axial <= .25 )
795 return a_inverse ? reversed_axial : axial;
796 else if( a_sagittal <= .25 )
797 return s_inverse ? reversed_sagittal : sagittal;
798 else if( a_coronal <= .25 )
799 return c_inverse ? reversed_coronal : coronal;
800 else
801 assert( false );
802
803 return axial; //will never be reached
804}

880size_t Image::spliceDownTo( dimensions dim ) //rowDim = 0, columnDim, sliceDim, timeDim
881{
882 if( lookup[0]->getRelevantDims() < ( size_t ) dim ) {
883 LOG( Debug, error ) << "The dimensionality of the chunks of this image is already below " << dim << " cannot spliceAt it.";
884 return 0;
885 } else if( lookup[0]->getRelevantDims() == dim ) {
886 LOG( Debug, info ) << "Skipping useless splicing, relevantDims is already " << lookup[0]->getRelevantDims();
887 return lookup.size();
888 }
889
890 // get a list of needed properties (everything which is missing in a newly created chunk plus everything which is needed for autosplice)
891 static const std::list<util::PropertyMap::PropPath> splice_needed{
892 "voxelSize","voxelGap","rowVec","columnVec","sliceVec","indexOrigin","acquisitionNumber"
893 };
894 static util::PropertyMap::PathSet needed = MemChunk<short>( 1 ).getMissing();
895 needed.insert( splice_needed.begin(), splice_needed.end() );
896 // reset the Chunk set, so we can insert new splices
897 set = _internal::SortedChunkList( defaultChunkEqualitySet );
898 set.addSecondarySort( "acquisitionNumber" );
899 set.addSecondarySort( "acquisitionTime" ); //@todo this should not be here
900
901 clean = false; // mark the image for reIndexing
902
903 //spliceAt existing lists into the current chunks -- they will be spliced further if necessary
904 PropertyMap::splice(lookup.begin(),lookup.end(),true);
905
906 //transfer properties needed for the chunk back into the chunk (if they're there but not lists)
907 for( util::PropertyMap::PathSet::const_reference need : needed ) {
908 const auto foundNeed=queryProperty( need );
909 if(foundNeed){
910 for( std::shared_ptr<Chunk> &ref : lookup ) {
911 if( !ref->hasProperty( need ) ) {
912 LOG( Debug, verbose_info ) << "Copying " << std::make_pair(need, *foundNeed) << " from the image to the chunk for splicing";
913 ref->touchProperty( need ) = *foundNeed;
914 } else
915 LOG(Debug,error) << need << " was found in the chunk although it is in the image as well. It will be deleted in the image";
916 }
917 this->remove(need);
918 }
919 }
920 // do the splicing
921 std::vector<std::shared_ptr<Chunk> > buffer;
922 buffer.swap(lookup); // move the old lookup table into a buffer, so it is empty when the splicer starts inserting the new chunks
923 for(auto &c_ptr:buffer){
924 for(auto &c:c_ptr->autoSplice(dim)){
925 const bool inserted=insertChunk(c);
926 assert(inserted);//it was in there, should go in again
927 }
928 }
929 reIndex();
930 return lookup.size();
931}
