85void PropertyValue::push_back( PropertyValue&& ref ){
86 if(!isEmpty() && getTypeID()!=ref.getTypeID())
87 {
88 LOG(Debug,info)
89 << "Inserting values of inconsistent type " << MSubject(ref.toString(true))
90 << " into " << MSubject(*this) << " trying to convert";
91 if(ref.fitsInto(this->getTypeID())) // ok, we can make the new data fit...
92 ref = ref.copyByID(this->getTypeID());
93 else if(this->fitsInto(ref.getTypeID())) // we can fit the old data fit the new one, not great, not terrible ...
94 container = this->copyByID(ref.getTypeID()).container;
95 else {
96 LOG(Debug,error) << "Inserting value of inconsistent type " << MSubject(ref.toString(true)) << " into " << MSubject(*this);
97 }
98 }
99 container.splice(container.end(),ref.container);
100}

165bool PropertyValue::transform(uint16_t dstID)
166{
167 PropertyValue ret,err;
168 for(const Value& ref : container){
169 #pragma message "no error handling for failed convert"
170 ret.push_back(ref.copyByID( dstID ));
171 }
172
173 if(!err.isEmpty()){
174 LOG( Debug, error ) << "Interpretation of " << err << " as " << util::getTypeMap().at(dstID) << " failed. Keeping old type.";
175 return false;
176 } else {
177 container.swap(ret.container);
178 return true;
179 }
180}

214std::vector< PropertyValue > PropertyValue::splice( const size_t len )
215{
216 assert(len);
217 size_t remain=size();//we use this to keep track of the number of remaining elements
218 std::vector<PropertyValue> ret(ceil(double(remain)/len));
219
220 for(PropertyValue &dst:ret){
221 auto e=container.begin();std::advance(e,std::min(len,remain));//move either len elements, or all that is left
222 dst.container.splice(dst.begin(),container,container.begin(),e);
223 remain-=dst.container.size();
224 assert(size()==remain);
225 }
226 assert(isEmpty());
227 return ret;
228}
229size_t PropertyValue::explode(size_t factor, const std::function<Value(const Value &)>& op)
230{
231 for(auto e=container.begin();e!=container.end();){
232 for(size_t i=0;i<factor;i++){
233 container.insert(e,op(*e));
234 }
235 e=container.erase(e);
236 }
237 return container.size();
238}

327template<typename OP>
328PropertyValue PropertyValue::operator_impl(const PropertyValue &rhs) const
329{
330 PropertyValue ret;
331 auto end=container.begin();
332 if(rhs.size()<size()){
333 LOG(Runtime,warning)
334 << "The PropertyValue at the left is loger than the one on the right. Will run operation "
335 << typeid(OP).name() << " only " << rhs.size() << " times";
336 std::advance(end,rhs.size());
337 } else
338 end=container.end();
339
340 LOG_IF(rhs.size()>size(),Runtime,warning)
341 << "The PropertyValue at the left is shorter than the one on the right. Will run operation "
342 << typeid(OP).name() << " only " << size() << " times";
343
344 auto ret_it= std::inserter(ret.container,ret.container.begin());
345 std::transform(container.begin(),container.end(),rhs.container.begin(),ret_it,OP{});
346 return ret;
347}
348template<typename OP>
349PropertyValue PropertyValue::operator_impl(const Value &rhs) const
350{
351PropertyValue ret;
352auto ret_it= std::inserter(ret.container,ret.container.begin());
353std::transform(container.begin(),container.end(),ret_it,std::bind(OP{},std::placeholders::_1,rhs));
354return ret;
355}
356void PropertyValue::resize(size_t size, const Value &insert)
357{
358 const auto mysize=container.size();
359 if(size-mysize>1){
360 LOG(Debug, warning ) << "Resizing a Property. You should avoid this, as it is expensive.";
361 }
362 if(mysize<size){ // grow
363 container.insert(container.end(), size-mysize, insert);
364 } else if(mysize>size){ // shrink
365 auto erasestart=std::next(container.begin(),size);
366 erase(erasestart,container.end());
367 }
368}
