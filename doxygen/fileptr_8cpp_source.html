67bool FilePtr::map( _internal::FileHandle &&file, size_t len, bool write, const std::filesystem::path &filename )
68{
69 void *ptr = nullptr;
70 FILE_HANDLE mmaph = 0;
71 rlimit rlim;
72 getrlimit(RLIMIT_DATA,&rlim);
73
74 if(rlim.rlim_cur<len){
75 if(rlim.rlim_max>len){
76 rlim.rlim_cur=len;
77 setrlimit(RLIMIT_NOFILE, &rlim);
78 } else {
79 LOG(Runtime,warning) << "Can't increase the limit for for mapped file size to " << len << ", this will crash..";
80 }
81 }
82 assert(file.good());
83
84#ifdef WIN32 //mmap is broken on Windows - workaround stolen from http://crupp.de/2007/11/14/howto-port-unix-mmap-to-win32/
85 mmaph = CreateFileMapping( file, 0, write ? PAGE_READWRITE : PAGE_WRITECOPY, 0, 0, NULL );
86
87 if( mmaph ) {
88 ptr = MapViewOfFile( mmaph, write ? FILE_MAP_WRITE : FILE_MAP_COPY, 0, 0, 0 );
89 }
90
91#else
92 ptr = mmap( 0, len, PROT_WRITE | PROT_READ, write ? MAP_SHARED : MAP_PRIVATE , file, 0 ); // yes we say PROT_WRITE here also if the file is opened ro - It's for the mapping, not for the file
93#endif
94
95 if( ptr == nullptr || ptr == reinterpret_cast<void*>(-1) ) {
96 switch(errno){
97 case ENOMEM:
98 LOG( Runtime, error ) << "Exceeded limit of mapping size or amount of mapped files";break;
99 default:
100 LOG( Runtime, error ) << "Failed to map " << util::MSubject( filename ) << ", error was " << util::getLastSystemError();
101 }
102 return false;
103 } else {
104 const Closer cl = {file.release(), mmaph, len, filename, write};
105 writing = write;
106 static_cast<ByteArray&>( *this ) = ByteArray( std::shared_ptr<uint8_t>(static_cast<uint8_t * const>( ptr ),cl), len );
107 return true;
108 }

111size_t FilePtr::checkSize(bool write, const std::filesystem::path &filename, size_t size)
112{
113 const auto currSize = std::filesystem::file_size( filename );
114 if ( std::numeric_limits<size_t>::max() < currSize ) {
115 LOG( Runtime, error )
116 << "Sorry cannot map files larger than " << std::numeric_limits<size_t>::max()
117 << " bytes on this platform";
118 return 0;
119 }
120
121 if( write ) { // if we're writing
122 assert( size > 0 );
123
124 if( size > currSize ) { // and the file is shorter than requested, resize it
125 std::error_code ec;
126 std::filesystem::resize_file(filename,size,ec);
127 if(ec){
128 LOG( Runtime, error )
129 << "Failed to resize " << util::MSubject( filename )
130 << " to the requested size " << size << ", the error was: " << util::MSubject( ec.message() );
131 return 0; // fail
132 }
133 }
134 return std::filesystem::file_size(filename);
135 } else { // if we're reading
136 if( size == 0 )
137 return currSize; // automatically select size of the file
138 else if( size <= currSize )
139 return size; // keep the requested size (will fit into the file)
140 else { // size will not fit into the file (and we cannot resize) => fail
141 LOG( Runtime, error ) << "The requested size for readonly mapping of " << util::MSubject( filename )
142 << " is greater than the filesize (" << currSize << ").";
143 return 0; // fail
144 }
145 }

148FilePtr::FilePtr(const std::filesystem::path &filename, size_t len, bool write, size_t mapsize)
149{
150 _internal::FileHandle file(filename,!write);
151 if(!file.good())
152 return;
153
154 const size_t file_size = checkSize(write, filename, len); // get the mapping size
155
156 if(file_size > mapsize || write) {
157 m_good = map(std::move(file), file_size, write, filename); //and do the mapping
158 LOG(Debug, verbose_info) << "Mapped " << file_size << " bytes of " << filename << " at " << getRawAddress().get();
159 }
160 else if(file_size>0)
161 {
162 static_cast<ValueArray*>(this)->operator=(ValueArray::make<uint8_t>(file_size));
163 if(!getRawAddress()){
164 LOG(Runtime,error) << "Failed creating memory for reading " << filename << ", the error was: " << util::getLastSystemError();
165 return;
166 }
167
168 size_t read_pos=0;
169 while (read_pos<file_size){
170 const auto result = read(file,getRawAddress(read_pos).get(),file_size);
171 if(result>0) {
172 read_pos += result;
173 } else if(result==0) {
174 LOG(Runtime,warning) << "Unexpected end reading " << filename << " (missing " << file_size-read_pos << " bytes)";
175 break;
176 } else {
177 LOG(Runtime, error) << "Error reading " << filename << ", the error was: " << util::getLastSystemError();
178 return;
179 }
180 }
181 m_good = true;
182 } else {
183 LOG(Runtime,error) << "Not opening empty file";
184 }
185 // from here on the pointer will be set if mapping succeeded
186}
