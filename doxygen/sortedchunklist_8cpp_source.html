34bool SortedChunkList::posCompare::operator()( const util::fvector3 &posA, const util::fvector3 &posB ) const
35{
36 if ( posA.lexical_less_reverse( posB ) ) { //if chunk is "under" the other - put it there
37 LOG( Debug, verbose_info ) << "Successfully sorted chunks by in-image position (" << posA << " below " << posB << ")";
38 return true;
39 }
40
41 return false;

43bool SortedChunkList::scalarPropCompare::operator()( const isis::util::PropertyValue &a, const isis::util::PropertyValue &b ) const
44{
45 const util::Value &aScal = a.front();
46 const util::Value &bScal = b.front();
47
48 if ( aScal.lt( bScal ) ) {
49 LOG( Debug, verbose_info ) << "Successfully sorted chunks by " << propertyName << " (" << aScal.toString( false ) << " before " << bScal.toString( false ) << ")";
50 return true;
51 } else
52 return false;
53}

103std::pair<std::shared_ptr<Chunk>, bool> SortedChunkList::secondaryInsert( SecondaryMap &map, const Chunk &ch )
104{
105 util::PropertyMap::key_type propName = map.key_comp().propertyName;
106 const auto found = ch.queryProperty( propName );
107
108 if( found ) {
109 //check, if there is already a chunk
110 std::shared_ptr<Chunk> &pos = map[*found];
111 bool inserted = false;
112
113 //if not. put ours there
114 if( !pos ) {
115 pos.reset( new Chunk( ch ) );
116 inserted = true;
117 }
118
119 assert( pos ); // here it must have some content
120 return std::make_pair( pos, inserted );
121 } else {
122 LOG( Runtime, warning ) << "Cannot insert chunk. It's lacking the property " << util::MSubject( propName ) << " which is needed for primary sorting";
123 return std::pair<std::shared_ptr<Chunk>, bool>( std::shared_ptr<Chunk>(), false );
124 }

126std::pair<std::shared_ptr<Chunk>, bool> SortedChunkList::primaryInsert( const Chunk &ch )
127{
128 static const util::PropertyMap::PropPath rowVecProb( "rowVec" ), columnVecProb( "columnVec" ), sliceVecProb( "sliceVec" ), indexOriginProb( "indexOrigin" );
129 LOG_IF( secondarySort.empty(), Debug, error ) << "There is no known secondary sorting left. Chunksort will fail.";
130 assert( ch.isValid() );
131 // compute the position of the chunk in the image space
132 // we don't have this position, but we have the position in scanner-space (indexOrigin)
133 const util::fvector3 origin = ch.getValueAs<util::fvector3>( indexOriginProb );
134 // and we have the transformation matrix
135 // [ rowVec ]
136 // [ columnVec]
137 // [ sliceVec]
138 // [ 0 0 0 1 ]
139 const util::fvector3 rowVec = ch.getValueAs<util::fvector3>( rowVecProb );
140 const util::fvector3 columnVec = ch.getValueAs<util::fvector3>( columnVecProb );
141 const util::fvector3 sliceVec = ch.getValueAsOr<util::fvector3>(
142 sliceVecProb,
143 util::fvector3({
144 rowVec[1] * columnVec[2] - rowVec[2] * columnVec[1],
145 rowVec[2] * columnVec[0] - rowVec[0] * columnVec[2],
146 rowVec[0] * columnVec[1] - rowVec[1] * columnVec[0]
147 })
148 );
149
150 // this is actually not the complete transform (it lacks the scaling for the voxel size), but it's enough
151 const util::fvector3 key{ origin.dot(rowVec), origin.dot(columnVec), origin.dot(sliceVec)};
152 const scalarPropCompare &secondaryComp = secondarySort.top();
153
154 // get the reference of the secondary map for "key" (create and insert a new if neccessary)
155 SecondaryMap &subMap = chunks.insert( std::make_pair( key, SecondaryMap( secondaryComp ) ) ).first->second;
156
157 // run insert on that
158 return secondaryInsert( subMap, ch ); // insert ch into the right secondary map
159}

162bool SortedChunkList::insert( const Chunk &ch )
163{
164 LOG_IF( secondarySort.empty(), Debug, error ) << "Inserting will fail without any secondary sort. Use chunks.addSecondarySort at least once.";
165 LOG_IF( !ch.isValid(), Debug, error ) << "You're trying insert an invalid chunk. The missing properties are " << ch.getMissing();
166 LOG_IF( !ch.isValid(), Debug, error ) << "You should definitively check the chunks validity (use the function Chunk::valid) before calling this function. Aborting now..";
167 assert( ch.isValid() );
168
169 if(isEmpty()){ // find secondary Sort from first chunk
170 std::stack<scalarPropCompare> backup = secondarySort;
171 while( !ch.hasProperty( secondarySort.top().propertyName ) ) {
172 const util::PropertyMap::key_type temp = secondarySort.top().propertyName;
173
174 if ( secondarySort.size() > 1 ) {
175 secondarySort.pop();
176 } else {
177 LOG( Debug, warning )
178 << "First chunk is missing the last secondary sort-property fallback (" << util::MSubject( temp ) << "), won't insert.";
179 secondarySort = backup;
180 return false;
181 }
182 }
183
184 LOG( Debug, info ) << "Using " << secondarySort.top().propertyName << " for secondary sorting, determined by the first chunk";
185 }
186
187 size_t sort_prop_size=ch.queryProperty(secondarySort.top().propertyName)->size();
188 if(sort_prop_size>1){ // secondary sort is multi value, we have to spliceAt the chunk and insert separately
189 LOG(Runtime,info) << "Splicing chunk at top dim as secondary sorting property " << secondarySort.top().propertyName << " is a list of size " << sort_prop_size;
190
191 // get rid of all not-to-be-splices props to save time
192 data::Chunk chs=ch;
193 auto extracted=chs.extract_if(
194 [sort_prop_size](const util::PropertyValue &p){
195 return p.size()!=sort_prop_size;
196 }
197 );
198
199 //ok, but some are actually needed for the splicing and inserting...
200 for(const auto &need:protected_props){
201 extracted.extract(need,chs);// .. so put them back
202 }
203
204 not_spliced.push_back( // store extracted with an (for now) empty list of chunks
205 std::pair< util::PropertyMap, std::list<std::shared_ptr<Chunk>> >(extracted,{})
206 );
207
208 LOG(Debug,info) << "Removed " << not_spliced.back() << " before splicing";
209
210 //find splicing depth
211 bool ok=true;
212
213 //figure out the splicing depth based on the length of the secondary sort
214 dimensions splicedim=timeDim;
215 for(
216 size_t prod = 1;
217 splicedim>=rowDim;
218 splicedim=static_cast<dimensions>(splicedim-1)
219 ){
220 prod *= chs.getDimSize(splicedim);
221 if(prod==sort_prop_size)
222 break;
223 else if(prod>sort_prop_size){
224 LOG(Runtime,error) << "Automatic splicing of chunk failed as amount of elements in splicing property does not fit amount of splices";
225 return false;
226 }
227 }
228
229 for(const data::Chunk &c:chs.autoSplice(splicedim)){
230 std::shared_ptr<Chunk> inserted=insert_impl(c);
231 if(inserted){
232 not_spliced.back().second.push_back(inserted); //list all chunks those extracted props belong into
233 } else {
234 ok=false;
235 }
236 }
237 if(!ok)
238 not_spliced.pop_back();
239 return ok;
240 } else {
241 return (bool)insert_impl(ch);
242 }
243}

245std::shared_ptr<Chunk> SortedChunkList::insert_impl(const Chunk &ch){
246 if( !isEmpty() ) {
247 // compare some attributes of the first chunk and the one which shall be inserted
248 const Chunk &first = *( chunks.begin()->second.begin()->second );
249
250 if ( first.getSizeAsVector() != ch.getSizeAsVector() ) { // if they have different size - do not insert
251 LOG( Debug, verbose_info )
252 << "Ignoring chunk with different size. (" << ch.getSizeAsString() << "!=" << first.getSizeAsString() << ")";
253 return nullptr;
254 }
255
256 for(const util::PropertyMap::PropPath & ref : equalProps ) { // check all properties which where given to the constructor of the list
257 // if at least one of them has the property, and they are not equal - do not insert
258 if ( first.hasProperty( ref ) && !(first.property( ref ) == ch.property( ref ) )) { //"==" will be false if ch.property is empty or different
259 LOG( Debug, verbose_info )
260 << "Ignoring chunk with different " << ref << ". Is " << util::MSubject( ch.property( ref ) )
261 << " but chunks already in the list have " << util::MSubject( first.property( ref ) );
262 return nullptr;
263 }
264 }
265 } else {
266 LOG( Debug, verbose_info ) << "Inserting 1st chunk";
267
268 if( ch.getDimSize( sliceDim ) > 1 ) {
269 LOG( Runtime, info ) << "We're dealing with volume chunks, considering indexOrigin as equal across Images";
270 equalProps.push_back( "indexOrigin" );
271 }
272
273 }
274
275 const std::pair<std::shared_ptr<Chunk>, bool> inserted = primaryInsert( ch );
276 if(inserted.first && !inserted.second){
277 LOG(Debug, info )
278 << "Not inserting chunk because there is already a Chunk at the same position (" << ch.property( "indexOrigin" )
279 << ") with the equal property " << std::make_pair( secondarySort.top().propertyName, ch.property( secondarySort.top().propertyName ) );
280
281 LOG_IF(
282 ch.queryProperty( "source" ) != std::const_pointer_cast<const Chunk>(inserted.first)->queryProperty( "source" ),//empty properties are not unequal
283 Debug, info )
284 << "The conflicting chunks where from " << ch.getValueAs<std::string>( "source" ) << " and " << inserted.first->getValueAs<std::string>( "source" );
285 }
286 return inserted.second ? inserted.first:nullptr;
287}

312size_t SortedChunkList::makeRectangular(util::slist* rejected)
313{
314 const std::set<size_t> images = getShape();//get lenghts of all primary sorted "columns" -- as set is sorted the smallest will be at begin
315 size_t dropped = 0;
316
317 if( images.size() > 1 ) { //cut down everything else if there is some
318 assert(chunks.begin()->second.begin()->second->getRelevantDims()<4); //images made of multiple 4D Chunks cannot be
319 size_t resize = *images.begin();
320
321 for(PrimaryMap::value_type &c:chunks) { // in every "column"
322 SecondaryMap &it = c.second;
323
324 if( it.size() > resize ) { //remove everything behind the shortest length
325 dropped += it.size() - resize;
326 SecondaryMap::iterator firstinvalid = it.begin();
327 std::advance( firstinvalid, resize );
328 if(rejected)
329 for(SecondaryMap::iterator i=firstinvalid;i!=it.end();i++)
330 rejected->push_back(i->second->getValueAs<std::string>("source"));
331 it.erase( firstinvalid, it.end() );
332 }
333
334 assert( it.size() == resize );
335 }
336
337 LOG_IF( dropped, Runtime, warning ) << "Dropped " << dropped << " chunks to make " << identify(true,false) << " rectangular";
338 }
339
340 return dropped;
341}

349std::vector< std::shared_ptr< Chunk > > SortedChunkList::getLookup()
350{
351 LOG_IF( getShape().size() != 1, Debug, error ) << "Running getLookup on an non rectangular chunk-list is not defined";
352
353 if( !isEmpty() ) {
354 PrimaryMap::iterator iP = chunks.begin();
355 const size_t horizontal = chunks.size();
356 const size_t vertical = iP->second.size();
357 std::vector< std::shared_ptr< Chunk > > ret( horizontal * vertical );
358
359 for( size_t h = 0; h < horizontal; h++, iP++ ) { // outer loop iterates horizontaly (through the primary sorting)
360 assert( iP != chunks.end() );
361 SecondaryMap::iterator iS = iP->second.begin();
362
363 for( size_t v = 0; v < vertical; v++, iS++ ) { // inner loop iterates verticaly (through the secondary sorting)
364 assert( iS != iP->second.end() );
365 ret[h + v * horizontal] = iS->second; // insert horizontally - primary sorting is the fastest running index (read the sorting matrix horizontaly)
366 }
367 }
368
369 return ret;
370 } else
371 return std::vector< std::shared_ptr< Chunk > >();
372}

382std::string SortedChunkList::identify(bool withpath, bool withdate, getproplist source, getproplist seqNum, getproplist seqDesc, getproplist seqStart) const
383{
384 forall(seqNum);
385 forall(seqDesc);
386
387 std::string ret;
388 if(seqNum.size()==1)
389 ret+=std::string("S")+seqNum.begin()->toString();
390 if(seqDesc.size()==1)
391 ret+= (ret.empty() ? std::string():std::string("_"))+seqDesc.begin()->toString();
392 if(withpath){
393 forall(source);
394 std::list<std::filesystem::path> sources;
395 std::transform(source.begin(),source.end(),std::back_inserter(sources),[](const util::PropertyValue &v){
396 return v.as<std::string>();
397 });
398 ret+=
399 (ret.empty() ? std::string():std::string(" "))+
400 ( std::string( "from " ) + util::getRootPath(sources,true).native() );
401 }
402 if(withdate){
403 forall(seqStart);
404 if(seqStart.size()==1)
405 ret+=(ret.empty() ? std::string():std::string(" "))+std::string("taken at ") + seqStart.begin()->toString();
406 }
407 return ret;
408}
